## P1

```c
#include <stdio.h>

void to_lower(char *s) {
    while(*s != 0) { // Aprovechando el cero del final del string
    if ('A' <= *s && *s <= 'Z') { // if (*s es mayuscula)
        *s += 'a' - 'A';
    }
    s++;
    }
}

void to_upper(char *s) {
    while(*s != 0) { // Aprovechando el cero del final del string
    if ('a' <= *s && *s <= 'z') { // if (*s es mayuscula)
        *s += 'A' - 'a';
    }
    s++;
    }
}

int main() {
    char s[] = "hOlAqeuEtal";
    to_lower(s);
    printf("%s\n", s);
    to_upper(s);
    printf("%s\n", s);
    return 0;
}
```
